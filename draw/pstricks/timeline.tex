\newcommand\BoxText[4][3cm]{%
  \savebox\TBox{\parbox{#1}{#4}}
  \rput[t](#2){\psframebox[fillstyle=gradient]{\usebox\TBox}}%
  \pnode(#2|0,0){A}\pnode(#2){B}
  \ncline{->}{A}{B}%
  \uput*{0.5cm}[-90](A){\footnotesize #3}%
}
